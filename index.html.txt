  <script>
    async function sendMessage() {
      const input = document.getElementById('userInput');
      const chat = document.getElementById('chat');
      const message = input.value;
      if (!message) return;

      chat.innerHTML += `<div class='message user'>You: ${message}</div>`;
      input.value = '';

      const response = await fetch('https://medical-bill-navigator-patient-rgriff-44dyp8dn6-med-course-pro.vercel.app/chat', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ message })
      });

      const data = await response.json();
      chat.innerHTML += `<div class='message'>Navigator: ${data.reply}</div>`;
      chat.scrollTop = chat.scrollHeight;
    }
  </script>
